/*
 Problem Statement:Write a program which accept matrix and check whether the matrix is Sparse 
                   matrix or not. Sparse matrix is a matrix with the majority of its elements 
                   equal to zero. 
Input : 1 0 3 0
        0 6 0 0
        0 0 1 0
        9 0 0 9
        
Output:True
*/
import java.lang.*;
import java.util.*;

class Matrix
{
   public boolean ChkSparse(int Arr[][],int Row,int Col)
   {
	   int i=0,j=0,iCnt=0;
	   for(i=0;i<Row;i++)
	   {
		   for(j=0;j<Col;j++)
		   {
			   if(Arr[i][j]==0)
			   {
				   iCnt++;
			   }
		   }
	   }
	   if(iCnt>(Row+Col))
	   {
		   return true;
	   }
	   else
	   {
		   return false;
	   }
	  
   }
}
     
class Demo
{
    public static void main(String []args)
    {
    	Scanner in=new Scanner(System.in);
    	Matrix Mobj=new Matrix();
    	int i=0,j=0;
    	System.out.println("Enter the number of rows:");
    	int iRow=in.nextInt();
    	System.out.println("Enter the number of columns:");
    	int iCol=in.nextInt();
    	int arr[][]=new int[iRow][iCol];
    	System.out.println("Enter element of matrix:");
    	for(i=0;i<arr.length;i++)
    	{
    		for(j=0;j<arr[i].length;j++)
    		{
    			arr[i][j]=in.nextInt();
    		}
    	}
    
    	boolean bRet=Mobj.ChkSparse(arr,iRow,iCol);
    	if(bRet)
    	{
    		System.out.println("It is Sparse Matix");
    	}
    	else
    	{
    		System.out.println("It is Not Sparse Matix");
    	}
    }
}
